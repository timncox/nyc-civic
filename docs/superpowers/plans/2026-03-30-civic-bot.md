> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a webhook-based MMP bot (`@nyc-civic`) that creates district-scoped group chats, onboards users via DM, broadcasts civic updates, and integrates with the NYC Civic dashboard as a Chat tab.

**Architecture:** Standalone Express service (`nyc-civic-bot/`) receives MMP webhooks, manages district groups via MMP's MCP API, pulls civic data from the nyc-civic MCP server. A lightweight MCP client class handles communication with both servers. The nyc-civic dashboard gets a Chat tab that proxies to MMP through new tools.

**Tech Stack:** TypeScript, Express, node-cron, better-sqlite3, MCP SDK (client), MMP (mmp.chat)

### Task 2: MCP client for calling MMP and nyc-civic tools

The bot needs to call MCP tools on two servers (MMP and nyc-civic) via HTTP. This is a lightweight client that initializes a StreamableHTTP session and calls tools.

**Files:**
- Create: `src/mcp-client.ts`

- [ ] **Step 1: Create `src/mcp-client.ts`**

```typescript
/**
 * Lightweight MCP client for calling tools on remote MCP servers.
 * Maintains a session via StreamableHTTP (initialize once, reuse).
 */

const TIMEOUT = 30_000;

interface McpToolResult {
  content: Array<{ type: string; text: string }>;
  isError?: boolean;
}

export class McpClient {
  private baseUrl: string;
  private sessionId: string | null = null;
  private nextId = 1;

  constructor(baseUrl: string) {
    this.baseUrl = baseUrl;
  }

  private async request(method: string, params?: unknown): Promise<unknown> {
    const headers: Record<string, string> = {
      "Content-Type": "application/json",
      "Accept": "application/json, text/event-stream",
    };
    if (this.sessionId) {
      headers["mcp-session-id"] = this.sessionId;
    }

    const res = await fetch(this.baseUrl, {
      method: "POST",
      headers,
      body: JSON.stringify({
        jsonrpc: "2.0",
        method,
        params: params ?? {},
        id: this.nextId++,
      }),
      signal: AbortSignal.timeout(TIMEOUT),
    });

    // Capture session ID from response headers
    const sid = res.headers.get("mcp-session-id");
    if (sid) this.sessionId = sid;

    if (!res.ok) {
      throw new Error(`MCP ${res.status}: ${await res.text()}`);
    }

    const data = (await res.json()) as any;
    if (data.error) {
      throw new Error(`MCP error: ${data.error.message}`);
    }
    return data.result;
  }

  async connect(): Promise<void> {
    await this.request("initialize", {
      protocolVersion: "2025-03-26",
      capabilities: {},
      clientInfo: { name: "nyc-civic-bot", version: "1.0.0" },
    });
  }

  async callTool(name: string, args: Record<string, unknown> = {}): Promise<McpToolResult> {
    if (!this.sessionId) await this.connect();
    const result = (await this.request("tools/call", { name, arguments: args })) as McpToolResult;
    return result;
  }

  /** Call a tool and parse the JSON text response */
  async call<T = unknown>(name: string, args: Record<string, unknown> = {}): Promise<T> {
    const result = await this.callTool(name, args);
    if (result.isError) {
      const msg = result.content?.[0]?.text ?? "Tool error";
      throw new Error(msg);
    }
    const text = result.content?.[0]?.text ?? "{}";
    return JSON.parse(text) as T;
  }
}
```

```typescript
import Database from "better-sqlite3";
import path from "node:path";

const DB_PATH = process.env.DB_PATH || path.join(process.cwd(), "data", "civic-bot.db");

let db: Database.Database;

export function getDb(): Database.Database {
  if (db) return db;

  const dir = path.dirname(DB_PATH);
  import("node:fs").then(fs => fs.mkdirSync(dir, { recursive: true })).catch(() => {});
  require("node:fs").mkdirSync(dir, { recursive: true });

  db = new Database(DB_PATH);
  db.pragma("journal_mode = WAL");

  db.exec(`
    CREATE TABLE IF NOT EXISTS user_districts (
      mmp_user_id TEXT NOT NULL,
      mmp_handle TEXT NOT NULL,
      address TEXT NOT NULL,
      council_district INTEGER,
      assembly_district INTEGER,
      election_district INTEGER,
      borough TEXT,
      lat REAL,
      lng REAL,
      joined_at INTEGER NOT NULL,
      PRIMARY KEY (mmp_user_id)
    );

    CREATE TABLE IF NOT EXISTS district_groups (
      district_key TEXT NOT NULL,
      tier TEXT NOT NULL,
      mmp_thread_id TEXT NOT NULL,
      district_number INTEGER,
      created_at INTEGER NOT NULL,
      PRIMARY KEY (district_key)
    );

    CREATE TABLE IF NOT EXISTS broadcast_log (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      district_key TEXT NOT NULL,
      broadcast_type TEXT NOT NULL,
      content_hash TEXT NOT NULL,
      sent_at INTEGER NOT NULL
    );
  `);

  return db;
}

// --- User districts ---

export interface UserDistrict {
  mmp_user_id: string;
  mmp_handle: string;
  address: string;
  council_district: number | null;
  assembly_district: number | null;
  election_district: number | null;
  borough: string | null;
  lat: number | null;
  lng: number | null;
  joined_at: number;
}

export function upsertUserDistrict(u: UserDistrict): void {
  getDb().prepare(`
    INSERT OR REPLACE INTO user_districts
    (mmp_user_id, mmp_handle, address, council_district, assembly_district, election_district, borough, lat, lng, joined_at)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `).run(u.mmp_user_id, u.mmp_handle, u.address, u.council_district, u.assembly_district, u.election_district, u.borough, u.lat, u.lng, u.joined_at);
}

export function getUserDistrict(mmpUserId: string): UserDistrict | null {
  return getDb().prepare("SELECT * FROM user_districts WHERE mmp_user_id = ?").get(mmpUserId) as UserDistrict | null;
}

// --- District groups ---

export interface DistrictGroup {
  district_key: string;
  tier: string;
  mmp_thread_id: string;
  district_number: number | null;
  created_at: number;
}

export function getDistrictGroup(key: string): DistrictGroup | null {
  return getDb().prepare("SELECT * FROM district_groups WHERE district_key = ?").get(key) as DistrictGroup | null;
}

export function upsertDistrictGroup(g: DistrictGroup): void {
  getDb().prepare(`
    INSERT OR REPLACE INTO district_groups
    (district_key, tier, mmp_thread_id, district_number, created_at)
    VALUES (?, ?, ?, ?, ?)
  `).run(g.district_key, g.tier, g.mmp_thread_id, g.district_number, g.created_at);
}

export function getGroupsByTier(tier: string): DistrictGroup[] {
  return getDb().prepare("SELECT * FROM district_groups WHERE tier = ?").all(tier) as DistrictGroup[];
}

// --- Broadcast log ---

export function hasBroadcast(districtKey: string, contentHash: string): boolean {
  const row = getDb().prepare("SELECT 1 FROM broadcast_log WHERE district_key = ? AND content_hash = ?").get(districtKey, contentHash);
  return !!row;
}

export function logBroadcast(districtKey: string, broadcastType: string, contentHash: string): void {
  getDb().prepare("INSERT INTO broadcast_log (district_key, broadcast_type, content_hash, sent_at) VALUES (?, ?, ?, ?)").run(districtKey, broadcastType, contentHash, Math.floor(Date.now() / 1000));
}
```

### Task 4: Onboarding logic — address resolution + group creation

**Files:**
- Create: `src/onboard.ts`

- [ ] **Step 1: Create `src/onboard.ts`**

This is the core logic: given a user handle and address, resolve districts, create/find MMP groups, and add the user.

```typescript
import type { McpClient } from "./mcp-client.js";
import {
  upsertUserDistrict, getDistrictGroup, upsertDistrictGroup,
  type UserDistrict, type DistrictGroup,
} from "./db.js";

interface DistrictInfo {
  council: number | null;
  communityBoard: string | null;
  stateSenate: number | null;
  stateAssembly: number | null;
  congressional: number | null;
  electionDistrict: number | null;
  borough: string | null;
  lat: number | null;
  lng: number | null;
}

interface OnboardResult {
  districts: DistrictInfo;
  groups: Array<{ key: string; tier: string; threadId: string; memberCount?: number; created: boolean }>;
  errors: string[];
}

/**
 * Onboard a user: resolve their address, create/find district groups, add them.
 */
export async function onboardUser(
  mmpClient: McpClient,
  civicClient: McpClient,
  senderHandle: string,
  senderUserId: string,
  address: string,
): Promise<OnboardResult> {
  const errors: string[] = [];
  const groups: OnboardResult["groups"] = [];

  // 1. Resolve address to districts via nyc-civic
  const districts = await civicClient.call<DistrictInfo>("lookup_address", { address });

  // 2. Save user district mapping
  const now = Math.floor(Date.now() / 1000);
  upsertUserDistrict({
    mmp_user_id: senderUserId,
    mmp_handle: senderHandle,
    address,
    council_district: districts.council,
    assembly_district: districts.stateAssembly,
    election_district: districts.electionDistrict,
    borough: districts.borough,
    lat: districts.lat,
    lng: districts.lng,
    joined_at: now,
  });

  // 3. Create/join district groups
  const districtKeys: Array<{ key: string; tier: string; number: number | null }> = [];

  if (districts.council != null) {
    districtKeys.push({ key: `CD-${districts.council}`, tier: "cd", number: districts.council });
  }
  if (districts.stateAssembly != null) {
    districtKeys.push({ key: `AD-${districts.stateAssembly}`, tier: "ad", number: districts.stateAssembly });
  }
  if (districts.stateAssembly != null && districts.electionDistrict != null) {
    const edKey = `ED-${districts.stateAssembly}-${String(districts.electionDistrict).padStart(3, "0")}`;
    districtKeys.push({ key: edKey, tier: "ed", number: districts.electionDistrict });
  }

  for (const dk of districtKeys) {
    try {
      let group = getDistrictGroup(dk.key);
      let created = false;

      if (!group) {
        // Create the MMP group
        const result = await mmpClient.call<{ thread_id: string; member_count: number }>(
          "mmp-create-group",
          { name: `NYC ${dk.key}`, members: [senderHandle] },
        );
        group = {
          district_key: dk.key,
          tier: dk.tier,
          mmp_thread_id: result.thread_id,
          district_number: dk.number,
          created_at: now,
        };
        upsertDistrictGroup(group);
        created = true;

        groups.push({ key: dk.key, tier: dk.tier, threadId: result.thread_id, memberCount: result.member_count, created });
      } else {
        // Group exists — add the user
        try {
          await mmpClient.call("mmp-add-member", { thread_id: group.mmp_thread_id, handle: senderHandle });
        } catch (e: any) {
          // May already be a member — that's fine
          if (!e.message?.includes("already in this group")) {
            errors.push(`${dk.key}: ${e.message}`);
          }
        }
        groups.push({ key: dk.key, tier: dk.tier, threadId: group.mmp_thread_id, created: false });
      }
    } catch (e: any) {
      errors.push(`${dk.key}: ${e.message}`);
    }
  }

  return { districts, groups, errors };
}
```

```typescript
import express from "express";
import { createHmac } from "crypto";
import { McpClient } from "./mcp-client.js";
import { getDb } from "./db.js";
import { onboardUser } from "./onboard.js";

// --- Config ---
const PORT = parseInt(process.env.PORT || "3002", 10);
const MMP_URL = process.env.MMP_URL || "https://mmp.chat/mcp";
const MMP_BOT_TOKEN = process.env.MMP_BOT_TOKEN || "";
const CIVIC_URL = process.env.CIVIC_URL || "http://localhost:3001/mcp";
const WEBHOOK_SECRET = process.env.WEBHOOK_SECRET || "";

// --- MCP Clients ---
const mmpClient = new McpClient(
  MMP_BOT_TOKEN ? `${MMP_URL}?token=${MMP_BOT_TOKEN}` : MMP_URL,
);
const civicClient = new McpClient(CIVIC_URL);

// --- Express ---
const app = express();
app.use(express.json());

// Health check
app.get("/health", (_req, res) => {
  res.json({ ok: true, bot: "@nyc-civic" });
});

// MMP Webhook receiver
app.post("/webhook", async (req, res) => {
  // Verify webhook signature if secret is configured
  if (WEBHOOK_SECRET) {
    const signature = req.headers["x-mmp-signature"] as string;
    const expected = createHmac("sha256", WEBHOOK_SECRET)
      .update(JSON.stringify(req.body))
      .digest("hex");
    if (signature !== expected) {
      res.status(401).json({ error: "Invalid signature" });
      return;
    }
  }

  const payload = req.body;
  console.log(`[webhook] ${payload.event} from @${payload.from_handle}`);

  // Handle message.received events
  if (payload.event === "message.received") {
    try {
      await handleIncomingMessage(payload);
    } catch (e) {
      console.error("[webhook] Error handling message:", e);
    }
  }

  res.json({ ok: true });
});

async function handleIncomingMessage(payload: {
  thread_id: string;
  from_handle: string;
  message_id: string;
  to_handle: string;
  [key: string]: unknown;
}): Promise<void> {
  // Fetch the actual message content from the thread
  const thread = await mmpClient.call<{
    type: string;
    messages: Array<{ from_handle: string; body: string | null; id: string }>;
  }>("mmp-thread", { thread_id: payload.thread_id });

  // Find the specific message
  const msg = thread.messages?.find((m) => m.id === payload.message_id);
  const body = msg?.body ?? "";

  if (thread.type === "dm") {
    // DM to the bot — treat as onboarding address
    await handleDM(payload.from_handle, body, payload.thread_id);
  } else {
    // Group @mention — handle commands
    await handleGroupMention(payload.from_handle, body, payload.thread_id);
  }
}

async function handleDM(fromHandle: string, body: string, threadId: string): Promise<void> {
  const address = body.trim();

  if (!address || address.length < 5) {
    await mmpClient.call("mmp-send", {
      to: `@${fromHandle}`,
      body: `Hi! Send me an NYC address and I'll add you to your neighborhood group chats.\n\nExample: 250 Broadway, New York, NY`,
      thread_id: threadId,
    });
    return;
  }

  try {
    // Use a placeholder user ID based on handle (MMP doesn't expose user IDs via webhook)
    const result = await onboardUser(mmpClient, civicClient, fromHandle, `user:${fromHandle}`, address);

    const groupLines = result.groups.map((g) => {
      const status = g.created ? "(new group!)" : "";
      return `  • NYC ${g.key} ${status}`;
    }).join("\n");

    const districtInfo = [
      result.districts.council != null && `Council District ${result.districts.council}`,
      result.districts.stateAssembly != null && `Assembly District ${result.districts.stateAssembly}`,
      result.districts.electionDistrict != null && `Election District ${result.districts.stateAssembly}-${result.districts.electionDistrict}`,
    ].filter(Boolean).join("\n  ");

    let reply = `Welcome! Found your districts:\n\n  ${districtInfo}\n\nYou've been added to ${result.groups.length} groups:\n${groupLines}`;

    if (result.errors.length > 0) {
      reply += `\n\nSome issues: ${result.errors.join("; ")}`;
    }

    await mmpClient.call("mmp-send", {
      to: `@${fromHandle}`,
      body: reply,
      thread_id: threadId,
    });
  } catch (e: any) {
    await mmpClient.call("mmp-send", {
      to: `@${fromHandle}`,
      body: `Sorry, I couldn't resolve that address. Make sure it's a valid NYC street address.\n\nError: ${e.message}`,
      thread_id: threadId,
    });
  }
}

async function handleGroupMention(fromHandle: string, body: string, threadId: string): Promise<void> {
  const lower = body.toLowerCase();

  if (lower.includes("help") || lower.includes("commands")) {
    await mmpClient.call("mmp-send", {
      thread_id: threadId,
      body: `@nyc-civic commands:\n• "stats" — group and district info\n• "reps" — your elected representatives\n• "help" — this message`,
    });
  }
  // Add more group commands as needed
}

});
```

```typescript
import { createHash } from "crypto";
import type { McpClient } from "./mcp-client.js";
import { getGroupsByTier, hasBroadcast, logBroadcast, type DistrictGroup } from "./db.js";

function contentHash(text: string): string {
  return createHash("sha256").update(text).digest("hex").slice(0, 16);
}

/**
 * Send a message to a district group, skipping if already sent (by content hash).
 */
async function broadcastToGroup(
  mmpClient: McpClient,
  group: DistrictGroup,
  broadcastType: string,
  message: string,
): Promise<boolean> {
  const hash = contentHash(message);
  if (hasBroadcast(group.district_key, hash)) return false;

  try {
    await mmpClient.call("mmp-send", {
      thread_id: group.mmp_thread_id,
      body: message,
    });
    logBroadcast(group.district_key, broadcastType, hash);
    return true;
  } catch (e) {
    console.error(`[broadcast] Failed for ${group.district_key}:`, e);
    return false;
  }
}

/**
 * Weekly 311 digest for all AD groups.
 */
export async function broadcast311Digest(
  mmpClient: McpClient,
  civicClient: McpClient,
): Promise<void> {
  const adGroups = getGroupsByTier("ad");
  console.log(`[broadcast] 311 digest for ${adGroups.length} AD groups`);

  for (const group of adGroups) {
    try {
      // We need lat/lng for the district — use a central point
      // For now, use the first user's coordinates in this district
      // TODO: store district centroid in district_groups table
      const complaints = await civicClient.call<{ complaints: any[]; total: number }>(
        "get_311",
        { lat: 40.7128, lng: -74.006, days_back: 7, limit: 50 }, // placeholder coords
      );

      if (complaints.total === 0) continue;

      // Aggregate by type
      const byCat: Record<string, number> = {};
      for (const c of complaints.complaints) {
        byCat[c.type] = (byCat[c.type] || 0) + 1;
      }
      const sorted = Object.entries(byCat).sort((a, b) => b[1] - a[1]);

      const lines = sorted.slice(0, 5).map(([type, count]) => `  • ${type}: ${count}`).join("\n");
      const message = `311 Digest — ${group.district_key}\n\nPast 7 days:\n${lines}\n\nTotal: ${complaints.total} complaints`;

      await broadcastToGroup(mmpClient, group, "311_digest", message);
    } catch (e) {
      console.error(`[broadcast] 311 failed for ${group.district_key}:`, e);
    }
  }
}

/**
 * Post-meeting attendance report for all CD groups.
 */
export async function broadcastAttendance(
  mmpClient: McpClient,
  civicClient: McpClient,
): Promise<void> {
  const cdGroups = getGroupsByTier("cd");
  console.log(`[broadcast] Attendance for ${cdGroups.length} CD groups`);

  try {
    const data = await civicClient.call<{
      meetings: number;
      summary: Array<{ name: string; present: number; absent: number; rate: string }>;
    }>("get_attendance", { meetings: 1 });

    if (!data.summary || data.summary.length === 0) return;

    const absent = data.summary.filter((s) => s.absent > 0);
    if (absent.length === 0) return;

    const absentLines = absent.map((a) => `  • ${a.name} — ABSENT`).join("\n");
    const message = `Attendance Report — Latest Stated Meeting\n\nAbsent members:\n${absentLines}\n\n${data.summary.length - absent.length} of ${data.summary.length} members present`;

    for (const group of cdGroups) {
      await broadcastToGroup(mmpClient, group, "attendance", message);
    }
  } catch (e) {
    console.error("[broadcast] Attendance failed:", e);
  }
}

/**
 * New legislation broadcast for CD groups.
 */
export async function broadcastNewLegislation(
  mmpClient: McpClient,
  civicClient: McpClient,
): Promise<void> {
  const cdGroups = getGroupsByTier("cd");

  try {
    const data = await civicClient.call<{ bills: any[]; total: number }>(
      "search_legislation",
      { query: "", level: "city" },
    );

    for (const bill of data.bills.slice(0, 10)) {
      const sponsors = bill.sponsors?.join(", ") || "Unknown";
      const message = `New Legislation\n\n${bill.id}: "${bill.title}"\nSponsored by: ${sponsors}\nStatus: ${bill.status || "Introduced"}`;

      // Broadcast to all CD groups (could be smarter — match sponsor to district)
      for (const group of cdGroups) {
        await broadcastToGroup(mmpClient, group, "legislation", message);
      }
    }
  } catch (e) {
    console.error("[broadcast] Legislation failed:", e);
  }
}
```
